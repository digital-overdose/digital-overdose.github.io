import { BASE_URL } from './../../data/url';
import { MetaService } from './../../services/meta.service';
import { Component, OnInit } from '@angular/core';

/**
 * Displays information on the Digital Overdose conference.
 *
 * @export
 */
@Component({
  selector: 'app-conference',
  templateUrl: './conference.component.html',
  styleUrls: ['./conference.component.scss']
})
export class ConferenceComponent implements OnInit {
  /**
   * Creates an instance of ConferenceComponent.
   *
   * @param meta The HTML header metadata injection service.
   */
  constructor(private meta: MetaService) { }

  /** Set the page metadata information. */
  ngOnInit(): void {
    this.meta.setTags(`${BASE_URL}/conference`, 'Digital Overdose Con: For Rookies',
                      `${BASE_URL}/assets/images/cover.jfif`, 750, 750,
                      ['conference', 'for', 'rookies'],
                      'Digital Overdose is proud to announce that we are running an information security conference !');
  }
}

<h2 class="text-center">(aka. D_OVER CON)</h2>

<br/>
<h3>What's all of this about ?</h3>
<p>Digital Overdose is proud to announce that we are running an information security conference !</p>

<p>The catch? The only people able to speak there are so-called rookies. Rookies not necessarily in the sense that they are new to Information Security, but more specifically in the sense that they've never had the experience of talking at a conference previously!</p>

<p>With the number of conventions with rookie tracks that have been cancelled, focusing specifically on rookies lets them have all of the well-deserved limelight, and helps foster the next generation of representatives of the community.</p>

<p>Anyone else is of course free to attend.</p>

<p>So if you're free on the 17th and 18th of April (2021), then <a target="_blank" href="https://calendar.google.com/event?action=TEMPLATE&tmeid=NzVuYjVxM2dlOHM5YzB0aWRhOXQ4MGluYzMgNDhyN3UxN2U1NWlmdDE5azc4M2Jidmw4bDhAZw&tmsrc=48r7u17e55ift19k783bbvl8l8%40group.calendar.google.com"><i class="far fa-calendar"></i> save the date</a> !</p>

<p>So we've managed to pique your interest ? That's awesome !</p>
